CREATE TABLE ms_student_tbl (
sid SERIAL NOT NULL,
fname VARCHAR(30) NOT NULL,
lname VARCHAR(30) NOT NULL,
phno VARCHAR(11),
degree_level VARCHAR(10),
year INT,
address VARCHAR(100),
PRIMARY KEY (sid)
);

room VARCHAR(10),
timings VARCHAR(100)
);
